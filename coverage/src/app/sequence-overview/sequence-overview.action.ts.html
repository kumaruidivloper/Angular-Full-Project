import { Action } from '@ngrx/store';
 
import {Filter, Sequence} from './sequence-overview.model';
import { PaginatedResponse } from '../core/interfaces/paginated-response.i';
import {PaginationParameters} from '../core/interfaces/pagination-params.i';
 
// @todo, we should apply the same refactoring here as was done in test-details.actions.ts
export const UPDATE_SEQUENCE_FILTER = '[SequenceOverview] Update Filters';
export const CLEAR_SEQUENCE_FILTERS = '[SequenceOverview] Clear Filters';
export const EMPTY_SEQUENCE_FILTERS = '[SequenceOverview] Empty Filters';
export const CLEAR_SEQUENCE_TABLE_FILTERS = '[SequenceOverview] Clear Filters of Sequence Table';
export const PAGINATE = '[SequenceOverview] Pages of Sequences';
export const LOAD_SEQUENCES = '[SequenceOverview] Load Sequences';
export const LOAD_SEQUENCES_SUCCESS = '[SequenceOverview] Load Sequences Successfully';
export const LOAD_SEQUENCES_FAILURE = '[SequenceOverview] Load Sequences Failure';
export const SORT = '[SequenceOverview] Sort Sequences';
export const SELECT_SEQUENCE = '[SequenceOverview] Select Sequences';
export const LOAD_ALL_SEQUENCE = '[SequenceOverview] Load All Sequence';
export const LOAD_TYPE_OF_SEQUENCE = '[SequenceOverview] Load Type of Sequence';
 
 
export class UpdateSequenceFilters implements Action {
  readonly type: string = UPDATE_SEQUENCE_FILTER;
  constructor(public payload: Filter) {}
}
 
export class ClearSequenceFilters implements Action {
  readonly type: string = CLEAR_SEQUENCE_FILTERS;
}
export class EmptySequenceFilters implements Action {
  readonly type: string = EMPTY_SEQUENCE_FILTERS;
}
export class ClearSequenceTableFilters implements Action {
  readonly type: string = CLEAR_SEQUENCE_TABLE_FILTERS;
}
 
export class LoadSequences implements Action {
  readonly type: string = LOAD_SEQUENCES;
}
 
export class PaginateSequence implements Action {
  readonly type: string = PAGINATE;
  constructor(public payload: PaginationParameters) {}
}
 
export class LoadSequencesSuccess implements Action {
  readonly type: string = LOAD_SEQUENCES_SUCCESS;
  constructor(public payload: PaginatedResponse<Sequence[]>) {}
}
 
export class LoadSequencesFailure implements Action {
  readonly type: string = LOAD_SEQUENCES_FAILURE;
}
 
export class SortSequence implements Action {
  readonly type: string = SORT;
  constructor(public payload: string) {}
}
 
export class SelectSequence implements Action {
  readonly type: string = SELECT_SEQUENCE;
  constructor(public sequence: Sequence) {}
}
 
export class LoadAllSequence implements Action {
  readonly type: string = LOAD_ALL_SEQUENCE;
}
 
export type SequenceOverviewActions = UpdateSequenceFilters | ClearSequenceFilters  | LoadSequences | LoadSequencesSuccess |
  LoadSequencesFailure | SortSequence | PaginateSequence | SelectSequence | LoadAllSequence | ClearSequenceTableFilters
  | EmptySequenceFilters;
