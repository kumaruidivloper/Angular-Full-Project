import { Injectable } from '@angular/core';
import { Actions, Effect } from '@ngrx/effects';
import { Observable } from 'rxjs/Observable';
import { Action } from '@ngrx/store';
 
import 'rxjs/add/operator/map';
import 'rxjs/add/operator/catch';
import 'rxjs/add/operator/switchMap';
import 'rxjs/add/operator/mergeMap';
import 'rxjs/add/observable/of';
import 'rxjs/add/operator/withLatestFrom';
 
import { RoutineOverviewService } from './routine-overview.service';
import { Routine } from './routine-overview.model';
import { PaginatedResponse } from '../core/interfaces/paginated-response.i';
import {
  CLEAR_ROUTINE_FILTERS, CLEAR_ROUTINE_TABLE_FILTERS, EmptyFilters, LOAD_ALL_ROUTINE, LOAD_ROUTINES,
  LOAD_ROUTINES_SUCCESS,
  LOAD_TYPE_OF_ROUTINE,
  LoadRoutines,
  LoadRoutinesFailure, PAGINATE,
  SORT,
  UPDATE_ROUTINE_FILTER
} from './routine-overview.action';
 
 
@Injectable()
 
export class RoutineOverviewEffects {
  constructor (private actions$: Actions, public routineOverviewService: RoutineOverviewService ) {}
 
  @Effect() getRoutineList$: Observable<Action> = this.actions$
    .ofType(LOAD_ROUTINES)
    .mergeMap((action) => {
      return this.routineOverviewService.getRoutineList()
        .map((result: PaginatedResponse<Routine[]>) => {
          // @todo creating actions using new ActionName()
          return {
            type: LOAD_ROUTINES_SUCCESS,
            payload: result
          };
        })
        .catch(() => {
          return Observable.of(new LoadRoutinesFailure());
        });
 
    });
 
  @Effect() changeParameters$: Observable<Action> = this.actions$
    .ofType(...[
      PAGINATE,
      UPDATE_ROUTINE_FILTER,
      SORT,
      LOAD_ALL_ROUTINE,
      LOAD_TYPE_OF_ROUTINE,
      CLEAR_ROUTINE_TABLE_FILTERS
    ])
    .map(() => {
      // @todo creating actions using new ActionName()
      return {
        type: LOAD_ROUTINES
      };
    });
  @Effect() clearFilters$: Observable<Action> = this.actions$
    .ofType(CLEAR_ROUTINE_FILTERS)
    .map(() => {
      return new EmptyFilters();
    });
}
