import { Action } from '@ngrx/store';
import { Credentials } from './login.model';
 
 
// @todo, we should apply the same refactoring here as was done in test-details.actions.ts
export const LOGIN = '[Auth] Login';
export const LOGOUT = '[Auth] Logout';
export const LOGIN_SUCCESS = '[Auth] Login Success';
export const LOGOUT_SUCCESS = '[Auth] Logout Success';
export const LOGIN_FAILURE = '[Auth] Login Failure';
export const TRY_LOGIN = '[Auth] Try Login';
export const TRY_LOGIN_SUCCESS = '[Auth] Try Login Success';
 
export class Login implements Action {
  readonly type: string = LOGIN;
  constructor(public credentials: Credentials) {}
}
 
export class Logout implements Action {
  readonly type: string = LOGOUT;
}
 
export class LoginSuccess implements Action {
  readonly type: string = LOGIN_SUCCESS;
  constructor(public token: string) {}
}
 
export class LogoutSuccess implements Action {
  readonly type: string = LOGOUT_SUCCESS;
}
 
export class LoginFailure implements Action {
  readonly type: string = LOGIN_FAILURE;
}
 
export class TryLogin implements Action {
  readonly type: string = TRY_LOGIN;
}
 
export class TryLoginSuccess implements Action {
  readonly type: string = TRY_LOGIN_SUCCESS;
  constructor(public token: string) {}
}
 
export type LoginActions = Login | LoginSuccess | Logout | LogoutSuccess | LoginFailure | TryLogin | TryLoginSuccess;
