import { Component, Input } from '@angular/core';
import { FormGroup } from '@angular/forms';
import { DynamicFormBase } from './dynamic-form-base';
 
 
@Component({
  selector: 'tm-dynamic-form',
  templateUrl: './dynamic-form.component.html'
})
export class DynamicFormComponent {
  @Input() formConfig: DynamicFormBase<any>;
  @Input() form: FormGroup;
  @Input() formValid: boolean;
  get isValid() {
    if (this.form.controls[this.formConfig.key].touched || this.form.controls[this.formConfig.key].dirty) {
      return this.form.controls[this.formConfig.key].valid;
    } else {
      return true;
    }
  }
 
}
