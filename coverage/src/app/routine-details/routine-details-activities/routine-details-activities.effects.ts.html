import { Injectable } from '@angular/core';
import { Actions, Effect } from '@ngrx/effects';
import { Action, Store } from '@ngrx/store';
import { cloneDeep } from 'lodash';
import { Observable } from 'rxjs/Observable';
import { PaginatedResponse } from '../../core/interfaces/paginated-response.i';
import { RoutineDetailsActivitiesService } from './routine-details-activities.service';
import { routineDetailsActionTypes, UpdateRoutineForm } from '../routine-details.action';
import { RoutineActivity } from '../routine-details.model';
import { routineDetailsSelector } from '../routine-details.reducer';
import {
  DELETE_ACTIVITIES,
  DeleteActivities,
  DESELECT_ACTIVITIES,
  DeselectActivities,
  GET_ACTIVITIES,
  LoadActivitiesFailure,
  LoadActivitiesSuccess,
  REORDER_ACTIVITIES,
  ReorderActivities,
  SELECT_ACTIVITIES,
  SelectActivities,
  UpdateSelectedRoutineActivities
} from './routine-details-activities.actions';
import { selectedRoutineActivitiesSelector, RoutineDetailsActivitiesState } from './routine-details-activities.reducer';
 
@Injectable()
export class RoutineDetailsActivitiesEffects {
  constructor(private actions$: Actions,
              private store: Store<RoutineDetailsActivitiesState>,
              private routineDetailsActivitiesService: RoutineDetailsActivitiesService) {
  }
 
  @Effect() selectActivities$: Observable<Action> = this.actions$
    .ofType(SELECT_ACTIVITIES)
    .withLatestFrom(this.store.select(selectedRoutineActivitiesSelector))
    .map(([action, selectedRoutineActivities]) => {
      const routineActivity = cloneDeep(selectedRoutineActivities);
      routineActivity.push((<SelectActivities>action).activities);
      return new UpdateSelectedRoutineActivities(routineActivity);
    });
 
  @Effect() deselectActivities$: Observable<Action> = this.actions$
    .ofType(DESELECT_ACTIVITIES)
    .withLatestFrom(this.store.select(selectedRoutineActivitiesSelector))
    .map(([action, selectedRoutineActivities]) => {
      const activitiesId = (<DeselectActivities>action).id;
      const routineActivity = selectedRoutineActivities.filter(activities => activities.id === activitiesId);
 
      return new UpdateSelectedRoutineActivities(routineActivity);
    });
 
  @Effect() getActivities$: Observable<Action> = this.actions$
    .ofType(GET_ACTIVITIES)
    .mergeMap(() => {
      return this.routineDetailsActivitiesService.getActivities()
        .map((result: PaginatedResponse<RoutineActivity[]>) => {
          return new LoadActivitiesSuccess(result);
        })
        .catch(() => {
          return Observable.of(new LoadActivitiesFailure());
        });
    });
 
  @Effect() changeParameters$: Observable<Action> = this.actions$
    .ofType(...[
      routineDetailsActionTypes.UPDATE_ACTIVITIES_FILTERS,
      routineDetailsActionTypes.CLEAR_ACTIVITIES_FILTERS,
      routineDetailsActionTypes.ACTIVITIES_PAGINATE
    ])
    .map(() => {
      return {
        type: GET_ACTIVITIES
      };
    });
 
  @Effect() reorderActivities$: Observable<Action> = this.actions$
    .ofType(REORDER_ACTIVITIES)
    .withLatestFrom(this.store.select(routineDetailsSelector))
    .map(([action, routineDetails]) => {
      const { from, to } = <ReorderActivities>action;
      const activities: RoutineActivity[] = routineDetails.objectCurrentRoutineVersion['routineActivity'];
      activities.splice(to, 0, activities.splice(from, 1)[0]);
 
      return new UpdateRoutineForm({
        ...routineDetails,
        routineActivity: activities
      });
    });
 
  @Effect() deleteActivities$: Observable<Action> = this.actions$
    .ofType(DELETE_ACTIVITIES)
    .withLatestFrom(this.store.select(routineDetailsSelector))
    .map(([action, routineDetails]) => {
      const activities = routineDetails.objectCurrentRoutineVersion['routineActivity'];
      activities.splice((<DeleteActivities>action).activities, 1);
 
      return new UpdateRoutineForm({
        ...routineDetails,
        routineActivity: activities
      });
    });
 
 
 
}
