import { RoutineActivity } from '../routine-details.model';
import { createFeatureSelector, createSelector } from '@ngrx/store';
import { SELECT_ACTIVITIES, GET_ACTIVITIES_SUCCESS, LoadActivitiesSuccess, RoutineDetailsActivitiesAction
, UPDATE_SELECTED_ROUTINE_ACTIVITIES, UpdateSelectedRoutineActivities
} from './routine-details-activities.actions';
import { PaginationParameters } from '../../core/interfaces/pagination-params.i';
import {SortOptions} from '../../core/interfaces/sort.i';
 
export const routineDetailsActivitiesFeatureName: string = 'RoutineDetailsActivities';
 
export interface RoutineDetailsActivitiesState {
  routineActivity: RoutineActivity[];
  selectedroutineActivity: RoutineActivity[];
  pagination: PaginationParameters;
}
 
export const defaultRoutineDetailsActivitiesState: RoutineDetailsActivitiesState = {
  routineActivity: [],
  selectedroutineActivity: [],
  pagination: {
    page: 1,
    pageSize: 5
  },
};
 
export function routineDetailsActivitiesReducer(state: RoutineDetailsActivitiesState = defaultRoutineDetailsActivitiesState,
                                             action: RoutineDetailsActivitiesAction): RoutineDetailsActivitiesState {
  switch (action.type) {
    case UPDATE_SELECTED_ROUTINE_ACTIVITIES:
      return {
        ...state,
        selectedroutineActivity: (<UpdateSelectedRoutineActivities>action).activities
      };
    case GET_ACTIVITIES_SUCCESS:
      return <RoutineDetailsActivitiesState>{
        ...state,
        routineActivity: (<LoadActivitiesSuccess> action).payload.data,
        pagination: (<LoadActivitiesSuccess> action).payload.pagination
      };
    default:
      return state;
  }
}
 
 
export const getRoutineActivities = (state: RoutineDetailsActivitiesState) => state.routineActivity;
export const getSelectedRoutineActivities = (state: RoutineDetailsActivitiesState) => state.selectedroutineActivity;
export const getRoutineActivitiesPagination = (state: RoutineDetailsActivitiesState) => state.pagination;
 
export const stateSelector = createFeatureSelector<RoutineDetailsActivitiesState>(routineDetailsActivitiesFeatureName);
export const routineActivitySelector = createSelector(stateSelector, getRoutineActivities);
export const selectedRoutineActivitiesSelector = createSelector(stateSelector, getSelectedRoutineActivities);
export const routineActivityPaginationSelector = createSelector(stateSelector, getRoutineActivitiesPagination);
