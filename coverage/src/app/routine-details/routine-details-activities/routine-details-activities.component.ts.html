import { Component, OnInit } from '@angular/core';
import { NgbModal, NgbModalRef } from '@ng-bootstrap/ng-bootstrap';
import { Store } from '@ngrx/store';
import { cloneDeep } from 'lodash';
import 'rxjs/add/operator/take';
import 'rxjs/add/operator/withLatestFrom';
import { Observable } from 'rxjs/Observable';
import { PaginationParameters } from '../../core/interfaces/pagination-params.i';
import { ClearActivitiesFilters, ActivitiesPaginate, UpdateRoutineForm, UpdateActivitiesFilters } from '../routine-details.action';
import { RoutineDetails, RoutineDetailsFilters, RoutineActivity, Filter } from '../routine-details.model';
import { filtersSelector, routineDetailsSelector} from '../routine-details.reducer';
import {
  DeselectActivities,
  DeleteActivities,
  LoadActivities,
  ReorderActivities,
  SelectActivities,
  UpdateSelectedRoutineActivities
} from './routine-details-activities.actions';
import {
  selectedRoutineActivitiesSelector,
  RoutineDetailsActivitiesState,
  routineActivitySelector,
  routineActivityPaginationSelector
} from './routine-details-activities.reducer';
import {async} from 'rxjs/scheduler/async';
 
@Component({
  selector: 'tm-routine-details-activities',
  templateUrl: './routine-details-activities.component.html',
  styleUrls: ['./routine-details-activities.component.scss']
})
export class RoutineDetailsActivitiesComponent implements OnInit {
 
  public routineDetails$: Observable<RoutineDetails>;
  public routineActivity$: Observable<RoutineActivity[]>;
  public selectedroutineActivity$: Observable<RoutineActivity[]>;
  public paginationParameters$: Observable<PaginationParameters>;
  public filters$: Observable<RoutineDetailsFilters>;
  private addActivitiesModal: NgbModalRef;
  public itemsPerPage = [5, 10, 15, 20];
 
  constructor(private store: Store<RoutineDetailsActivitiesState>,
              private ngModal: NgbModal) { }
 
  ngOnInit() {
    this.routineDetails$ = this.store.select(routineDetailsSelector);
    this.routineActivity$ = this.store.select(routineActivitySelector);
    this.selectedroutineActivity$ = this.store.select(selectedRoutineActivitiesSelector);
    this.paginationParameters$ = this.store.select(routineActivityPaginationSelector);
    this.filters$ = this.store.select(filtersSelector);
 
 
    this.store.dispatch(new LoadActivities());
  }
 
  onRoutineActivitiesRowSelect(activities, event) {
    if (event.target.checked) {
      this.store.dispatch(new SelectActivities(activities));
    } else {
      this.store.dispatch(new DeselectActivities(activities.id));
    }
  }
 
  saveActivities() {
    this.selectedroutineActivity$
      .first()
      .withLatestFrom(this.routineDetails$ )
      .subscribe(([selectedActivities, routineDetails]) => {
        this.store.dispatch(new UpdateRoutineForm({
          ...routineDetails,
         objectCurrentRoutineVersion : {
          ...routineDetails.objectCurrentRoutineVersion,
            routineActivity: [
              ...(routineDetails.objectCurrentRoutineVersion.routineActivity || []),
              ...selectedActivities
            ]
         }
        }));
        this.addActivitiesModal.close();
      });
  }
 
  open(template) {
    this.addActivitiesModal = this.ngModal.open(template, {
      size: 'lg',
    });
    this.store.dispatch(new UpdateSelectedRoutineActivities([]));
  }
 
  onPaginationChange(paginationParameters) {
    this.store.dispatch(new ActivitiesPaginate(paginationParameters));
  }
 
  onFilterChange(filter: Filter) {
    let filteredVal = {
        field: filter.field,
        value: filter.value
      };
    this.store.dispatch(new UpdateActivitiesFilters(filteredVal));
  }
 
  clearRoutineActivitiesFilters(): void {
    this.store.dispatch(new ClearActivitiesFilters());
  }
 
  onMoveUpClick(index) {
    this.store.dispatch(new ReorderActivities(index, index - 1));
  }
 
  onMoveDownClick(index) {
    this.store.dispatch(new ReorderActivities(index, index + 1));
  }
 
  onDeleteClick(index) {
    this.store.dispatch(new DeleteActivities(index));
  }
 
}
