import { Action } from '@ngrx/store';
import { RoutineActivity } from '../routine-details.model';
import { PaginatedResponse } from '../../core/interfaces/paginated-response.i';
 
export const SELECT_ACTIVITIES = '[RoutineDetailsActivities] Select Activities';
export const DESELECT_ACTIVITIES = '[RoutineDetailsActivities] Deselect Activities';
export const UPDATE_SELECTED_ROUTINE_ACTIVITIES = '[RoutineDetailsActivities] Update Selected Routine Activities';
export const GET_ACTIVITIES = '[RoutineDetailsActivities] Get Activities';
export const GET_ACTIVITIES_SUCCESS = '[RoutineDetailsActivities] Get Activities Success';
export const GET_ACTIVITIES_FAILURE = '[RoutineDetailsActivities] Get Activities Failure';
export const REORDER_ACTIVITIES = '[RoutineDetailsActivities] Reorder Activities';
export const DELETE_ACTIVITIES = '[RoutineDetailsActivities] Delete Activities';
 
 
export class SelectActivities implements Action {
  readonly type: string = SELECT_ACTIVITIES;
  constructor(public activities: RoutineActivity) {}
}
 
export class DeselectActivities implements Action {
  readonly type: string = DESELECT_ACTIVITIES;
  constructor(public id: number) {}
}
 
export class UpdateSelectedRoutineActivities implements Action {
  readonly type: string = UPDATE_SELECTED_ROUTINE_ACTIVITIES;
  constructor(public activities: RoutineActivity[]) {}
}
 
export class LoadActivities implements Action {
  readonly type: string = GET_ACTIVITIES;
}
 
export class LoadActivitiesSuccess implements Action {
  readonly type: string = GET_ACTIVITIES_SUCCESS;
  constructor(public payload: PaginatedResponse<RoutineActivity[]>) {}
}
 
export class LoadActivitiesFailure implements Action {
  readonly type: string = GET_ACTIVITIES_FAILURE;
}
 
export class ReorderActivities implements Action {
  readonly type: string = REORDER_ACTIVITIES;
  constructor(public from: number, public to: number) {}
}
 
export class DeleteActivities implements Action {
  readonly type: string = DELETE_ACTIVITIES;
 
  constructor(public activities: number) {}
}
 
export type RoutineDetailsActivitiesAction = SelectActivities | DeselectActivities | UpdateSelectedRoutineActivities
  | LoadActivities | LoadActivitiesSuccess | LoadActivitiesFailure;
