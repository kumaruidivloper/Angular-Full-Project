import {HttpClient, HttpParams} from '@angular/common/http';
import { Injectable } from '@angular/core';
import 'rxjs/add/operator/map';
import { Observable } from 'rxjs/Observable';
import {ResultType, TestCaseDetails, TruckFunction, TruckFunctionArea} from './test-case-detail.model';
 
@Injectable()
export class TestCaseDetailService {
  public url = 'testCase/';
  public resultTypeUrl = 'testCase/resultTypeTestCase';
  public truckFunctionAreaUrl = 'testCase/truckFunctionArea';
  public truckFunctionUrl = 'testCase/truckFunction';
  constructor(public http: HttpClient) { }
 
  public getTestCaseDetails(id): Observable<TestCaseDetails> {
    return this.http.get<TestCaseDetails>(this.url + id);
  }
  public getResultType(): Observable<ResultType> {
    return this.http.get<ResultType>(this.resultTypeUrl);
  }
  public getTruckFunctionArea(): Observable<TruckFunctionArea> {
    return this.http.get<TruckFunctionArea>(this.truckFunctionAreaUrl);
  }
  public getTruckFunction(id): Observable<TruckFunction> {
    console.log(id);
    let parameters: HttpParams;
    parameters = new HttpParams()
      .set('id', id);
    return this.http.get<TruckFunction>(this.truckFunctionUrl, {params: parameters});
  }
}
