import { Component, OnInit } from '@angular/core';
import {FormArray, FormBuilder, FormControl, FormGroup, Validators} from '@angular/forms';
import { ActivatedRoute } from '@angular/router';
import {Observable} from 'rxjs/Observable';
import { cloneDeep, isEqual, remove, uniq, uniqWith } from 'lodash';
import {ResultType, TestCaseDetails, TruckFunction, TruckFunctionArea} from './test-case-detail.model';
import {Store} from '@ngrx/store';
 
import {
  GetResultTypeTestCase, GetTestCaseDetails, GetTruckFunction,
  GetTruckFunctionArea
} from './test-case-detail.actions';
import {
  tcTruckFunctionAreaSelector, tcTruckFunctionSelector, testCaseDetailsSelector, TestCaseDetailsState,
  testCaseResultTypeSelector
} from './test-case-detail.reducer';
import {userSelector} from '../core/services/user/user.reducer';

import {SessionStorageService} from '../core/storage/session-storage.service';
 
@Component({

})

  private testCaseForm: FormGroup;
  public isCreate: boolean = true;
  public testCaseDetails$: Observable<TestCaseDetails>;
  public testCaseLevel = [1, 2, 3];
  public testCaseStatus = ['Active', 'Inactive'];
  public categoryOptions = ['STANDARD', 'DEVELOPMENT'];
  public testCasePrivateOptions = [{ id: 'Yes', value: 'true'}, {id: 'No', value: 'false'}];
  public speedUnits = ['km/h', 'ml/h'];
  public user$: Observable<User>;
  public site$: Observable<string>;
  public group$: Observable<string>;
  public resultType$: Observable<ResultType[]>;
  public truckFunctionArea$: Observable<TruckFunctionArea[]>;
  public truckFunction$: Observable<TruckFunction[]>;
  private getUserId: string;
  public currentDate = new Date();
  constructor(
    private route: ActivatedRoute,
    private store: Store<TestCaseDetailsState>,
    private formBuilder: FormBuilder,
    private sessionStorageService: SessionStorageService) { }
  onTestTeamGroup(value) {
    // this.store.dispatch(new UpdateUserGroup(value));

  ngOnInit() {
    this.user$ = this.store.select(userSelector);
    // this.site$ = this.store.select(updateUserSiteSelector);
    // this.group$ = this.store.select(updateUserGroupSelector);
    this.resultType$ =  this.store.select(testCaseResultTypeSelector);
    this.truckFunctionArea$ = this.store.select(tcTruckFunctionAreaSelector);
    this.truckFunction$ = this.store.select(tcTruckFunctionSelector);
    this.sessionStorageService.getItem('Login:UserId').subscribe( res => {
      this.getUserId = (res) ? res.toString() : '';
    });
    this.testCaseForm =  this.formBuilder.group({
      id: 0,
      testCaseType: 'TEST_CASE',
      level: '1',
      testCaseSite: '',
      testCaseUserGroup: '',
      testCaseVersion: [],
      currentTestCaseVersion:  new FormGroup({
        category: new FormControl('STANDARD', Validators.required),
        changeInfo: new FormControl(''),
        description: new FormControl(''),
        expectedResult: new FormControl(''),
        id: new FormControl(0),
        image: new FormControl(''),
        name: new FormControl(''),
        notes: new FormControl(''),
        privateTestCase: new FormControl('false', Validators.required),
        reqId: new FormControl(''),
        reqVersion: new FormControl(''),
        speed: new FormGroup({
          speedValue: new FormControl(''),
          speedUnit: new FormControl('km/h')
        }),
        status: new FormControl('Active', Validators.required),
        testCaseProductClassFilter:  new FormControl(''),
        testCaseResultTypeTC:  new FormControl({name: 'Done/Not Done'}),
        testCaseTag:  new FormControl(''),
        testCaseTools:  new FormControl(''),
        versionNo:  new FormControl('')

      testCaseTruckFunction: new FormControl('')

      this.store.dispatch(new GetResultTypeTestCase());
      this.store.dispatch(new GetTruckFunctionArea());
 
 
  }
  onChangeTruckFunctionArea(event): void {
   const truckAreaID = event.target.value;
    this.store.dispatch(new GetTruckFunction(truckAreaID));
    this.testCaseForm.controls['testCaseTruckFunction'].enable();
  }
  onSave(value): void {
    if (this.testCaseForm.valid) {
      let changeInfo;
      this.testCaseForm.markAsPristine();
      this.user$.subscribe( res => {
        changeInfo = {
          changedByFirstName: res.firstName || null,
          changedByLastName: res.lastName || null,
          changedByUserId: this.getUserId,
          // created: this.isCreate ? this.currentDate.getTime() : this.routineInfo['created'],
          lastChanged: this.currentDate.getTime()
        };
      });
      this.testCaseForm.patchValue({currentTestCaseVersion: {changeInfo: changeInfo}});
      this.testCaseForm.controls['testCaseVersion'].setValue([this.testCaseForm.controls['currentTestCaseVersion'].value]);
      console.log(this.testCaseForm.value);
    } else {
      const formControlList = this.testCaseForm.controls['currentTestCaseVersion'] as FormGroup;
      Object.keys( formControlList.controls).forEach(key => {
        formControlList.controls[key].markAsTouched({ onlySelf: true });
      });
    }
  }
  getField(field: string) {
    return this.testCaseForm.get(field);
  }
 
  compare(option: any, value: any) {
    return isEqual(option, value);
  }
 
  get name() { return this.getField('currentTestCaseVersion.name'); }
  get testCaseTruckFunction() { return this.getField('currentTestCaseVersion.testCaseTruckFunction'); }
 
  private findInvalidControls() {
    const invalid = [];
    const controls = this.testCaseForm.controls;
    for (const name in controls) {
      if (controls[name].invalid) {
        this.testCaseForm.controls[name].markAsTouched();
        invalid.push(name);
      }
    }
    return invalid;
  }
