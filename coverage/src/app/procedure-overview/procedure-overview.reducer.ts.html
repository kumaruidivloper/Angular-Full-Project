import { createFeatureSelector, createSelector } from '@ngrx/store';
import { cloneDeep, omitBy, pick } from 'lodash';
import { PaginationParameters } from '../core/interfaces/pagination-params.i';
import { SortDirection, SortOptions } from '../core/interfaces/sort.i';
import { UserGroup, UserSite } from '../core/services/user/user.model';
import {
  CLEAR_PROCEDURE_FILTERS,
  CLEAR_PROCEDURE_TABLE_FILTERS,
  LOAD_PROCEDURES_FAILURE,
  LOAD_PROCEDURES_SUCCESS,
  LoadProceduresSuccess,
  PAGINATE,
  Paginate,
  ProcedureOverviewActions,
  SELECT_PROCEDURE,
  SelectProcedure,
  SORT,
  Sort,
  UPDATE_PROCEDURE_FILTER,

  UPDATE_SELECTED_PROCEDURE_SITE,
  UpdateProcedureFilters,
  UpdateSelectedProcedureGroup,
  UpdateSelectedProcedureSite
} from './procedure-overview.actions';
import { Filter, Procedure, ProcedureOverviewFilters } from './procedure-overview.model';
 
 
export const procedureOverviewFeatureName = 'ProcedureOverview';
 
 
export interface ProcedureOverviewState {
  procedures: Procedure[];
  filters: ProcedureOverviewFilters;
  pagination: PaginationParameters;
  sort?: SortOptions;
  selectedProcedure: Procedure;
  getSelectedProcedureSite: UserSite;
  getSelectedProcedureGroup: UserGroup;
 
}
 
 
export const procedureOverviewDefaultState: ProcedureOverviewState = {
  procedures: [],
  filters: {},
  pagination: {
    page: 1,
    pageSize: 20
  },
  selectedProcedure: null,
  getSelectedProcedureSite: null,
  getSelectedProcedureGroup: null
};
 
// @todo according to the latest advice from DRS the logic of how to update the state should be moved to effects layer.
// the reducer will then simply update the state with the new value
// please review this reducer and apply changes if needed
export function procedureOverviewReducer (
  state: ProcedureOverviewState = procedureOverviewDefaultState,
  action: ProcedureOverviewActions): ProcedureOverviewState {
  const procedureWithOutType = '';
 
  switch (action.type) {
    case LOAD_PROCEDURES_FAILURE:
      return <ProcedureOverviewState>{
        ...state,
        procedures: procedureOverviewDefaultState.procedures,
        pagination: procedureOverviewDefaultState.pagination
      };
 
    case LOAD_PROCEDURES_SUCCESS:
      return <ProcedureOverviewState>{
        ...state,
        procedures: (<LoadProceduresSuccess> action).payload.data,
        pagination: (<LoadProceduresSuccess> action).payload.pagination
      };
 
    case UPDATE_PROCEDURE_FILTER:
      const filterToUpdate: Filter = (<UpdateProcedureFilters> action).payload;
 
      const updatedFilters: {[key: string]: string} = cloneDeep(state.filters);
 
      updatedFilters[filterToUpdate.field] = filterToUpdate.value;
 
      return <ProcedureOverviewState>{
        ...state,
        filters: omitBy(updatedFilters, value => value === '')
      };
 
    case SORT:
      const sort = getNextSort((<Sort>action).payload, state.sort);
 
      return <ProcedureOverviewState>{
        ...state,
        sort: sort
      };
 
 
    case PAGINATE:
      return <ProcedureOverviewState>{
        ...state,
        pagination: (<Paginate>action).payload
      };
 
    case CLEAR_PROCEDURE_TABLE_FILTERS:
      const filtersToRetain = pick(state.filters, ['siteId', 'groupId']);
      return <ProcedureOverviewState>{
        ...state,
        filters: filtersToRetain
      };
    case CLEAR_PROCEDURE_FILTERS:
      return <ProcedureOverviewState>{
        ...state,
        filters: {}
      };
 
    case SELECT_PROCEDURE:
      return {
        ...state,
        selectedProcedure: cloneDeep((<SelectProcedure>action).procedure)
      };
 
    case UPDATE_SELECTED_PROCEDURE_SITE:
      return {
        ...state,
        getSelectedProcedureSite: (<UpdateSelectedProcedureSite>action).site
      };
 
    case UPDATE_SELECTED_PROCEDURE_GROUP:
      return {
        ...state,
        getSelectedProcedureGroup: (<UpdateSelectedProcedureGroup>action).group
      };
 
    default:
      return state;
 
  }
 
}
 
 
export function getNextSort(field: string, sort: SortOptions): SortOptions | undefined {
  if (sort && sort.field === field) {
    const nextSort: SortDirection = (sort.direction + 1) % 3;
    return SortDirection[nextSort] && {
      field: field,
      direction: nextSort
    };
  }
 
  return {
    field: field,
    direction: SortDirection.ASC
  };
}
