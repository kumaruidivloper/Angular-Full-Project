import { Injectable } from '@angular/core';
import { Actions, Effect } from '@ngrx/effects';
import { Observable } from 'rxjs/Observable';
import { Action, Store } from '@ngrx/store';
 
import 'rxjs/add/operator/map';
import 'rxjs/add/operator/catch';
import 'rxjs/add/operator/switchMap';
import 'rxjs/add/operator/mergeMap';
import 'rxjs/add/observable/of';
import { stateSelector, TestOverviewState } from './test-overview.reducer';
 
import { TestOverviewService } from './test-overview.service';
import { Test } from './test-overview.model';
import { PaginatedResponse } from '../core/interfaces/paginated-response.i';
import {
  CLEAR_FILTERS, CLEAR_TABLE_FILTERS, EmptyFilters, LOAD_TESTS, LOAD_TESTS_SUCCESS, LoadTestsFailure,
  PAGINATE, SORT,
  UPDATE_FILTER, UPDATE_SELECTED_TEST_GROUP, UPDATE_SELECTED_TEST_SITE
} from './test-overview.actions';
 
@Injectable()
export class TestOverviewEffects {
  constructor(
    private actions$: Actions,
    public testOverViewService: TestOverviewService) {
  }
 
  @Effect() getTestList$: Observable<Action> = this.actions$
    .ofType(LOAD_TESTS)
    .mergeMap(() => {
      return this.testOverViewService.getTestList()
        .map((result: PaginatedResponse<Test[]>) => {
          // @todo creating actions using new ActionName()
         return {
            type : LOAD_TESTS_SUCCESS,
            payload: result
          };
        })
        .catch(() => {
          return Observable.of(new LoadTestsFailure());
        });
    });
 
  @Effect() changeParameters$: Observable<Action> = this.actions$
    .ofType(...[
      PAGINATE,
      UPDATE_FILTER,
      SORT,
      UPDATE_SELECTED_TEST_GROUP,
      UPDATE_SELECTED_TEST_SITE,
      CLEAR_TABLE_FILTERS
    ])
    .map(() => {
      // @todo creating actions using new ActionName()
      return {
        type: LOAD_TESTS
      };
    });
  @Effect() clearFilters$: Observable<Action> = this.actions$
    .ofType(CLEAR_FILTERS)
    .map(() => {
      return new EmptyFilters();
    });
}
