import { Component, OnInit } from '@angular/core';
import { Store } from '@ngrx/store';
import { Credentials } from './login.model';
import { LoginState } from './login.reducer';
import {SecurityState, tokenSelector} from '../../core/services/security/security.reducer';
import {Login, TryLogin} from '../../core/services/security/security.actions';
import { LoginForDev } from './login.actions';
import {environment} from '../../../environments/environment';
import {Router} from '@angular/router';
import {GetUser, GetUserGroups, GetUserRole, GetUserSites} from '../../core/services/user/user.actions';
 
@Component({
  selector: 'tm-login',
  templateUrl: './login.component.html',
  styleUrls: ['./login.scss']
})
export class LoginComponent implements OnInit {
  public showForm: boolean;
  public isTokenAvaialble$: any;
 
  constructor(private store: Store<LoginState>,
              private pingStore: Store<SecurityState>,
              private router: Router) {}
 
  public login(form) {
    const credentials: Credentials = {
      username: form.username.trim(),
      password: form.password.trim()
    };
    this.store.dispatch(new LoginForDev(credentials));
  }
 
  public showLoginForm(): void {
    // const getDomain =  window.location.hostname;
    // if (getDomain === `${environment.qaUrl}`) {
      this.isTokenAvaialble$.subscribe( res => {
        if (!res) {
          this.showForm = false;
          this.pingStore.dispatch(new Login());
        } else {
          this.store.dispatch(new TryLogin());
          this.store.dispatch(new GetUser());
          this.store.dispatch(new GetUserSites());
          this.store.dispatch(new GetUserGroups());
          this.store.dispatch(new GetUserRole());
          this.router.navigate(['/']);
        }
      });
    // } else {
    //   this.showForm = true;
    // }
  }
 
  ngOnInit() {
    this.isTokenAvaialble$ = this.pingStore.select(tokenSelector);
    this.showForm = false;
  }
}
