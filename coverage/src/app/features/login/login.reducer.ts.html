import {
  LOGIN_SUCCESS, LoginActions, LoginForDevSuccess, LOGOUT_SUCCESS,
  TRY_LOGIN_SUCCESS
} from './login.actions';
import { createFeatureSelector, createSelector } from '@ngrx/store';
 
export const loginFeatureName: string = 'Login';
 
export interface LoginState {
  token: string;
}
 
export const loginDefaultState: LoginState = {
  token: null
};
 
export function loginReducer(
  state: LoginState = loginDefaultState,
  action: LoginActions): LoginState {
 
  switch (action.type) {
    case LOGOUT_SUCCESS:
      return {
        ...state,
        token: null
      };
    case LOGIN_SUCCESS:
    case TRY_LOGIN_SUCCESS:
      return {
        ...state,
        token: (<LoginForDevSuccess>action).token
      };
    default:
      return state;
  }
}
 
export const getToken = state => state.token;
 
export const stateSelector = createFeatureSelector<LoginState>(loginFeatureName);
export const tokenSelector = createSelector(stateSelector, getToken);
